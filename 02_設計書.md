## 1. システムアーキテクチャ

### 1.1 技術スタック

| レイヤー | 技術 | 選定理由 |
|---|---|---|
| **フロントエンド** | Next.js 15 (App Router) | SSR/SSG対応、SEO最適化、React Server Components |
| **UIライブラリ** | Tailwind CSS + shadcn/ui | 高速開発、モダンなUI、カスタマイズ性 |
| **チャートライブラリ** | Recharts | React向け、カスタマイズ性、軽量 |
| **言語** | TypeScript | 型安全性、開発効率、計算ロジックの信頼性 |
| **バックエンド** | Next.js API Routes + Server Actions | フルスタック一体化、デプロイ簡素化 |
| **ORM** | Prisma | 型安全なDB操作、マイグレーション管理 |
| **データベース** | PostgreSQL (Supabase) | RLS、認証機能内蔵、無料枠あり |
| **認証** | NextAuth.js (Auth.js) v5 | Google OAuth、JWT |
| **AI** | OpenAI API (GPT-4o) | レポート生成、高品質な文章出力 |
| **PDF生成** | @react-pdf/renderer | React コンポーネントでPDF構築、グラフ埋め込み可能 |
| **決済** | Stripe | 単品購入 + サブスクリプション対応 |
| **メール** | Resend | トランザクションメール、添付ファイル対応 |
| **為替API** | ExchangeRate-API（無料枠）| 150+通貨、月1,500リクエスト無料 |
| **デプロイ** | Vercel | Next.jsとの親和性、エッジ配信、無料枠 |
| **CI/CD** | GitHub Actions | 自動テスト、自動デプロイ |
| **監視** | Vercel Analytics + Sentry | パフォーマンス・エラー監視 |
| **広告** | Google AdSense | 表示広告による収益化 |

### 2.2 テーブル定義

#### users（ユーザー）
```sql
CREATE TABLE users (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email           VARCHAR(255) UNIQUE NOT NULL,
  name            VARCHAR(100),
  avatar_url      VARCHAR(500),
  locale          VARCHAR(5) DEFAULT 'en',  -- 'en' | 'ja'
  plan            VARCHAR(20) DEFAULT 'free', -- 'free' | 'pro' | 'premium'
  email_verified  BOOLEAN DEFAULT FALSE,
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);
```

#### simulations（シミュレーション）
```sql
CREATE TABLE simulations (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID REFERENCES users(id) ON DELETE SET NULL, -- NULLは未ログインユーザー
  title           VARCHAR(200) DEFAULT 'Untitled Simulation',

  -- 入力データ（JSONB で柔軟に保存）
  input_data      JSONB NOT NULL,
  /*
    input_data の構造:
    {
      "country_from": "JP",
      "country_to": "SG",
      "income_current": 5000000,
      "income_target": 60000,
      "currency_current": "JPY",
      "currency_target": "SGD",
      "savings": 2000000,
      "savings_currency": "JPY",
      "rent_current": 80000,
      "rent_target": 2500,
      "living_cost_current": 120000,
      "living_cost_target": 1500,
      "tax_rate_current": 0.30,
      "tax_rate_target": 0.15,
      "exchange_rate": 110.5,
      "inflation_current": 0.025,
      "inflation_target": 0.02,
      "salary_growth": 0.02,
      "investment_return": 0.05,
      "simulation_years": 5
    }
  */

  -- 計算結果（JSONB で保存）
  result_data     JSONB,
  /*
    result_data の構造:
    {
      "annual_savings_current": 1200000,
      "annual_savings_target": 25000,
      "yearly_assets": [
        { "year": 0, "current": 2000000, "target": 18140 },
        { "year": 1, "current": 3200000, "target": 43140 },
        ...
      ],
      "real_assets": [...],
      "fire_age": 52,
      "monte_carlo": { ... },
      "sensitivity": { ... }
    }
  */

  -- 共有
  share_token     VARCHAR(32) UNIQUE, -- URLシェア用のユニークトークン
  is_public       BOOLEAN DEFAULT FALSE,

  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_simulations_user ON simulations(user_id);
CREATE INDEX idx_simulations_share ON simulations(share_token) WHERE share_token IS NOT NULL;
```

#### reports（AIレポート）
```sql
CREATE TABLE reports (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id             UUID REFERENCES users(id) ON DELETE SET NULL,
  simulation_id       UUID REFERENCES simulations(id) ON DELETE CASCADE,

  grade               VARCHAR(20) NOT NULL, -- 'basic' | 'standard' | 'premium'
  status              VARCHAR(20) DEFAULT 'pending',
                      -- 'pending' | 'generating' | 'completed' | 'failed'

  -- AI生成コンテンツ
  report_content      TEXT,            -- Markdown形式のレポート本文
  pdf_url             VARCHAR(500),    -- 生成されたPDFのURL

  -- 決済
  stripe_payment_id   VARCHAR(255),    -- Stripe PaymentIntent ID
  amount              DECIMAL(10,2),   -- 支払金額（USD）

  -- メール送信
  email_sent          BOOLEAN DEFAULT FALSE,
  email_sent_at       TIMESTAMPTZ,

  created_at          TIMESTAMPTZ DEFAULT NOW(),
  updated_at          TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_reports_user ON reports(user_id);
CREATE INDEX idx_reports_simulation ON reports(simulation_id);
```

#### subscriptions（サブスクリプション）
```sql
CREATE TABLE subscriptions (
  id                      UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id                 UUID UNIQUE REFERENCES users(id) ON DELETE CASCADE,
  stripe_customer_id      VARCHAR(255) UNIQUE,
  stripe_subscription_id  VARCHAR(255) UNIQUE,
  plan                    VARCHAR(20) NOT NULL, -- 'pro' | 'premium'
  status                  VARCHAR(20) NOT NULL, -- 'active' | 'canceled' | 'past_due' | 'trialing'
  current_period_start    TIMESTAMPTZ,
  current_period_end      TIMESTAMPTZ,
  cancel_at_period_end    BOOLEAN DEFAULT FALSE,
  created_at              TIMESTAMPTZ DEFAULT NOW(),
  updated_at              TIMESTAMPTZ DEFAULT NOW()
);
```

#### country_presets（国別プリセット）
```sql
CREATE TABLE country_presets (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  country_code        VARCHAR(3) UNIQUE NOT NULL, -- ISO 3166-1 alpha-2
  country_name        JSONB NOT NULL,             -- { "en": "Japan", "ja": "日本" }
  currency            VARCHAR(3) NOT NULL,        -- ISO 4217
  currency_symbol     VARCHAR(5),                 -- "¥", "$", "€" etc.
  default_tax_rate    DECIMAL(5,4) NOT NULL,      -- 0.3000 = 30%
  default_inflation   DECIMAL(5,4) NOT NULL,      -- 0.0250 = 2.5%
  reference_data      JSONB,
  /*
    {
      "rent": { "city_center": 80000, "outside_center": 55000 },
      "living_cost": { "single": 120000, "couple": 200000, "family": 300000 },
      "notes": { "en": "Tax includes income tax + social insurance", "ja": "税率は所得税+社会保険料込み" }
    }
  */
  is_active           BOOLEAN DEFAULT TRUE,
  updated_at          TIMESTAMPTZ DEFAULT NOW()
);
```

#### exchange_rate_cache（為替レートキャッシュ）
```sql
CREATE TABLE exchange_rate_cache (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  base_currency   VARCHAR(3) NOT NULL,
  target_currency VARCHAR(3) NOT NULL,
  rate            DECIMAL(18,8) NOT NULL,
  fetched_at      TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(base_currency, target_currency)
);

CREATE INDEX idx_exchange_rate_pair ON exchange_rate_cache(base_currency, target_currency);
```

// lib/simulation/basic-calculator.ts

export class BasicCalculator {
  calculate(input: SimulationInput): SimulationResult {
    const yearlyResults: YearlyResult[] = [];

    let assetCurrent = this.convertToBaseCurrency(
      input.currentSavings, input.savingsCurrency, input.currencyCurrent, input.exchangeRate
    );
    let assetTarget = this.convertToBaseCurrency(
      input.currentSavings, input.savingsCurrency, input.currencyTarget, input.exchangeRate
    );

    for (let year = 0; year <= input.simulationYears; year++) {
      const growthMultiplier = Math.pow(1 + input.salaryGrowthRate, year);
      const inflationCurrentMultiplier = Math.pow(1 + input.inflationCurrent, year);
      const inflationTargetMultiplier = Math.pow(1 + input.inflationTarget, year);

      // 収入（昇給考慮）
      const incomeCurrent = input.incomeCurrent * growthMultiplier;
      const incomeTarget = input.incomeTarget * growthMultiplier;

      // 支出（インフレ考慮）
      const expensesCurrent =
        (input.rentCurrent * inflationCurrentMultiplier +
         input.livingCostCurrent * inflationCurrentMultiplier) * 12;
      const expensesTarget =
        (input.rentTarget * inflationTargetMultiplier +
         input.livingCostTarget * inflationTargetMultiplier) * 12;

      // 税金
      const taxCurrent = incomeCurrent * input.taxRateCurrent;
      const taxTarget = incomeTarget * input.taxRateTarget;

      // 年間貯蓄
      const netSavingsCurrent = incomeCurrent - taxCurrent - expensesCurrent;
      const netSavingsTarget = incomeTarget - taxTarget - expensesTarget;

      // 資産（投資リターン + 貯蓄）
      if (year > 0) {
        assetCurrent = assetCurrent * (1 + input.investmentReturn) + netSavingsCurrent;
        assetTarget = assetTarget * (1 + input.investmentReturn) + netSavingsTarget;
      }

      // 実質資産（インフレ調整）
      const realAssetCurrent = assetCurrent / inflationCurrentMultiplier;
      const realAssetTarget = assetTarget / inflationTargetMultiplier;

      yearlyResults.push({
        year,
        incomeCurrent, expensesCurrent, taxCurrent,
        netSavingsCurrent, assetCurrent, realAssetCurrent,
        incomeTarget, expensesTarget, taxTarget,
        netSavingsTarget, assetTarget, realAssetTarget,
        assetTargetConverted: assetTarget * input.exchangeRate,
      });
    }

    return {
      input,
      baseCurrency: input.currencyCurrent,
      annualSavingsCurrent: yearlyResults[1]?.netSavingsCurrent ?? 0,
      annualSavingsTarget: yearlyResults[1]?.netSavingsTarget ?? 0,
      fiveYearAssetCurrent: yearlyResults[input.simulationYears]?.assetCurrent ?? 0,
      fiveYearAssetTarget: yearlyResults[input.simulationYears]?.assetTarget ?? 0,
      assetDifference: /* 計算 */,
      yearlyResults,
      monthlyBreakdown: /* 計算 */,
    };
  }
}

### 5.3 モンテカルロシミュレーション

```typescript
// lib/simulation/monte-carlo.ts

export class MonteCarloSimulator {
  private trials: number = 1000;

  simulate(input: SimulationInput, fxVolatility: number = 0.10): MonteCarloResult {
    const allFinalAssets: number[] = [];
    const yearlyPercentileData: number[][] = []; // [year][trial]

    // 各年のデータ初期化
    for (let y = 0; y <= input.simulationYears; y++) {
      yearlyPercentileData.push([]);
    }

    for (let trial = 0; trial < this.trials; trial++) {
      let asset = input.currentSavings;
      let fxRate = input.exchangeRate;

      for (let year = 0; year <= input.simulationYears; year++) {
        // 為替の確率的変動（幾何ブラウン運動）
        const z = this.standardNormal();
        fxRate = fxRate * Math.exp(-0.5 * fxVolatility * fxVolatility + fxVolatility * z);

        // 投資リターンの変動
        const returnZ = this.standardNormal();
        const actualReturn = input.investmentReturn + 0.15 * returnZ; // σ=15%

        // インフレの変動
        const infZ = this.standardNormal();
        const actualInflation = input.inflationTarget + 0.01 * infZ;

        // 年間貯蓄計算
        const growthMultiplier = Math.pow(1 + input.salaryGrowthRate, year);
        const inflationMultiplier = Math.pow(1 + actualInflation, year);

        const income = input.incomeTarget * growthMultiplier;
        const expenses = (input.rentTarget * inflationMultiplier +
                         input.livingCostTarget * inflationMultiplier) * 12;
        const tax = income * input.taxRateTarget;
        const netSavings = income - tax - expenses;

        if (year > 0) {
          asset = asset * (1 + actualReturn) + netSavings;
        }

        yearlyPercentileData[year].push(asset);
      }

      allFinalAssets.push(asset);
    }

    // パーセンタイル計算
    const percentiles = {
      p10: yearlyPercentileData.map(y => this.percentile(y, 10)),
      p25: yearlyPercentileData.map(y => this.percentile(y, 25)),
      p50: yearlyPercentileData.map(y => this.percentile(y, 50)),
      p75: yearlyPercentileData.map(y => this.percentile(y, 75)),
      p90: yearlyPercentileData.map(y => this.percentile(y, 90)),
    };

    return {
      trials: this.trials,
      years: input.simulationYears,
      percentiles,
      finalAssetDistribution: allFinalAssets.sort((a, b) => a - b),
      probabilityOfLoss: allFinalAssets.filter(a => a < input.currentSavings).length / this.trials,
    };
  }

  private standardNormal(): number {
    // Box-Muller 変換
    const u1 = Math.random();
    const u2 = Math.random();
    return Math.sqrt(-2 * Math.log(u1)) * Math.cos(2 * Math.PI * u2);
  }

  private percentile(arr: number[], p: number): number {
    const sorted = [...arr].sort((a, b) => a - b);
    const index = Math.ceil((p / 100) * sorted.length) - 1;
    return sorted[Math.max(0, index)];
  }
}

   ↓
④ Webhook: customer.subscription.created
   → subscriptions テーブルに INSERT
   → users テーブルの plan を更新
   ↓
⑤ Premium機能がアンロック

--- 毎月の更新 ---
⑥ Webhook: invoice.paid
   → subscriptions テーブルの current_period_end を更新

--- キャンセル ---
⑦ ユーザーがダッシュボードで解約
   → POST /api/subscription/cancel
   → Stripe で cancel_at_period_end = true
   → 期間終了後に Webhook: customer.subscription.deleted
   → users.plan = 'free' に更新
```

### 8.3 Stripe Webhook 処理

```typescript
// app/api/payment/webhook/route.ts

export async function POST(req: Request) {
  const body = await req.text();
  const sig = req.headers.get('stripe-signature')!;

  const event = stripe.webhooks.constructEvent(body, sig, webhookSecret);

  switch (event.type) {
    case 'checkout.session.completed':
      const session = event.data.object;
      if (session.mode === 'payment') {
        // 単品レポート購入 → レポート生成開始
        await handleReportPurchase(session);
      } else if (session.mode === 'subscription') {
        // サブスク開始
        await handleSubscriptionCreated(session);
      }
      break;

    case 'invoice.paid':
      // サブスク更新
      await handleSubscriptionRenewal(event.data.object);
      break;

    case 'customer.subscription.deleted':
      // サブスク終了
      await handleSubscriptionCanceled(event.data.object);
      break;
  }

  return new Response('OK', { status: 200 });
}
```
